phpMyAdmin

Créer la base de données

CREATE DATABASE tifosi;

creation du l'user tifosie avec les droit d'admin

CREATE USER 'tifosi'@'localhost' IDENTIFIED BY 'pass123!';
    GRANT ALL PRIVILEGES ON tifosi.* TO 'tifosi'@'localhost';
    FLUSH PRIVILEGES;
